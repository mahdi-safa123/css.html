<!-- مثال 2: الحسابات -->
<div class="example">
  <h2>2️⃣ إجراء عمليات حسابية</h2>
  <input type="number" id="num1" placeholder="الرقم الأول">
  <input type="number" id="num2" placeholder="الرقم الثاني">
  <button onclick="sumNumbers()">احسب المجموع</button>
  <p id="result"></p>
</div>

<!-- مثال 3: التحقق من شرط -->
<div class="example">
  <h2>3️⃣ التحقق من شرط</h2>
  <input type="number" id="score" placeholder="اكتب الدرجة">
  <button onclick="checkScore()">تحقق</button>
  <p id="scoreResult"></p>
</div>

<!-- مثال 4: تغيير لون الخلفية -->
<div class="example">
  <h2>4️⃣ تغيير لون خلفية الصفحة</h2>
  <button onclick="changeBackground()">تغيير اللون</button>
</div>

<!-- مثال 5: العد التنازلي -->
<div class="example">
  <h2>5️⃣ العد التنازلي</h2>
  <p id="countdown">5</p>
  <button onclick="startCountdown()">ابدأ العد</button>
</div>

<script>
  // مثال 1
  function changeText() {
    document.getElementById("text1").innerHTML = "أهلاً بك في JavaScript!";
  }

  // مثال 2
  function sumNumbers() {
    let a = parseFloat(document.getElementById("num1").value);
    let b = parseFloat(document.getElementById("num2").value);
    let sum = a + b;
    document.getElementById("result").innerHTML = "المجموع: " + sum;
  }

  // مثال 3
  function checkScore() {
    let score = parseFloat(document.getElementById("score").value);
    let result = score >= 50 ? "ناجح ✅" : "راسب ❌";
    document.getElementById("scoreResult").innerHTML = result;
  }

  // مثال 4
  function changeBackground() {
    const colors = ["lightblue","lightgreen","lightpink","lightyellow","lavender"];
    document.body.style.backgroundColor = colors[Math.floor(Math.random()*colors.length)];
  }

  // مثال 5
  function startCountdown() {
    let count = 5;
    const elem = document.getElementById("countdown");
    elem.innerHTML = count;
    const interval = setInterval(() => {
      count--;
      elem.innerHTML = count;
      if(count <= 0) clearInterval(interval);
    }, 1000);
  }
</script>
